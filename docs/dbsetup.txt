#lage tabeller:
CREATE TABLE patient_info(
patient_ID VARCHAR(10) PRIMARY KEY NOT NULL,
clinical_info TEXT,
family_ID INT,
sex CHAR(1) CHECK(Sex in ('M', 'F')),
panel INT,
FOREIGN KEY(panel) REFERENCES panels(ID)
);

CREATE TABLE panels(
ID INT PRIMARY KEY,
panel_name VARCHAR(30) NOT NULL
);

CREATE TABLE QC(
SAMPLE_NAME VARCHAR(10),
GENOME_SIZE INT,
BAIT_TERRITORY INT,
TARGET_TERRITORY INT,
BAIT_DESIGN_EFFICIENCY REAL,
TOTAL_READS INT,
PF_READS INT,
PF_UNIQUE_READS INT,
PCT_PF_READS INT,
PCT_PF_UQ_READS REAL,
PF_UQ_READS_ALIGNED REAL,
PCT_PF_UQ_READS_ALIGNED INT,
PF_UQ_BASES_ALIGNED INT,
ON_BAIT_BASES INT,
NEAR_BAIT_BASES INT,
OFF_BAIT_BASES INT,
ON_TARGET_BASES INT,
PCT_SELECTED_BASES REAL,
PCT_OFF_BAIT REAL,
ON_BAIT_VS_SELECTED REAL,
MEAN_BAIT_COVERAGE REAL,
MEAN_TARGET_COVERAGE REAL,
PCT_USABLE_BASES_ON_BAIT REAL,
PCT_USABLE_BASES_ON_TARGET REAL,
FOLD_ENRICHMENT REAL,
ZERO_CVG_TARGETS_PCT REAL,
FOLD_80_BASE_PENALTY REAL,
PCT_TARGET_BASES_2X REAL,
PCT_TARGET_BASES_10X REAL,
PCT_TARGET_BASES_20X REAL,
PCT_TARGET_BASES_30X REAL,
PCT_TARGET_BASES_40X REAL,
PCT_TARGET_BASES_50X REAL,
PCT_TARGET_BASES_100X REAL,
HS_LIBRARY_SIZE INT,
HS_PENALTY_10X REAL,
HS_PENALTY_20X REAL,
HS_PENALTY_30X REAL,
HS_PENALTY_40X REAL,
HS_PENALTY_50X INT,
HS_PENALTY_100X INT,
AT_DROPOUT INT,
GC_DROPOUT INT,
FOREIGN KEY(SAMPLE_NAME) REFERENCES patient_info(patient_ID),
);

CREATE TABLE runs(
sbs VARCHAR(12) PRIMARY KEY,
date DATETIME
);

CREATE TABLE raw_variants(
ID INT PRIMARY KEY,
chr VARCHAR(5),
start INT,
stop INT,
ref VARCHAR(50),
alt VARCHAR(50),
hg VARCHAR(10) DEFAULT 'hg19',
UNIQUE(Chr, Start, Stop, ref, alt)
);
